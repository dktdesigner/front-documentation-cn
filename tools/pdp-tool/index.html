# API列表

url type params desc
/api/users/login post code 通过code获取token
/api/users/current get - 返回当前登录用户信息
/api/product/search post model 根据model获取产品信息
/api/product/copy post dsm,platform, from,copy 复制指定平台数据
/api/product/getOrigin post model 返回range100原始数据
/api/product/update post dsm,platform,json 更新数据
/api/product/getPicture post model 获取图片

# api / 登录

// 登录
this.$store.dispatch('user/login', code)

// store
login({ commit, getters }, code) {
  return this.$axios
    .$post('/api/users/login', { code })
    .then(({ token }) => {
      console.log(token)
      if (token) {
        commit('init', token)
        this.$cookies.set('token', token, { path: '/', maxAge: 60 * 60 * 10 })
      }
      return getters.isLogin
    })
    .catch((e) => {
      console.log(e)
    })
}

# api / 获取当前用户

// 获取当前用户
this.$store.dispatch('user/current')

// store
current() {
  return this.$axios
    .$get('/api/users/current')
    .then((res) => {
      return res
    })
    .catch(() => {
      return { error: 'Unauthorized' }
    })
}
